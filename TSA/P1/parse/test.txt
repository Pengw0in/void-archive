  if(distance <= OBSTACLE_THRESHOLD && distance != 999) {
    stop();
    moveBackward();
    delay(100);
    stop();
    
    int leftDistance = scanDirection(180);
    int rightDistance = scanDirection(20);
    
    if(leftDistance > rightDistance && leftDistance >= MIN_SAFE_DISTANCE) {
      turnLeft();
      delay(TURN_DURATION);
    } else if(rightDistance >= MIN_SAFE_DISTANCE) {
      turnRight();
      delay(TURN_DURATION);
    }
    stop();
  } else {
    moveForward();
  }
}

void bluetoothControl() {
  if(Serial.available() > 0) {

void executeCommand(char cmd) {
  switch(cmd) {
    case 'F': case '^': moveForward(); break;

    case 'L': case '<': turnLeft(); break;
    case 'R': case '>': turnRight(); break;
    case 'S': case '*': stop(); break;
  }
}
